\column{.32\textwidth}
\begin{block}{学习者建模}
\small 从自然对话抽取 8 维画像，以行为证据持续修正难度与策略。\par\vspace{3mm}
\tagbox{accentred}{增量画像}\quad\tagbox{accentred}{动态推荐}
\end{block}
\end{columns}
\vspace{6mm}
\begin{center}

\end{center}

\section{工程特色 / 创新}

\begin{frame}{可信不是附加项，而是生成链路的组成部分}
\begin{columns}[T,onlytextwidth]
\column{.32\textwidth}
\begin{block}{输入侧}
\small 浏览器与服务端双层拦截危险操作性请求，并防范提示注入。
\end{block}
\column{.32\textwidth}
\begin{block}{生成侧}
\small 资源进入结果池前审核；绝对化结论、论文与外链标记待复核。
\end{block}
\column{.32\textwidth}
\begin{block}{证据侧}
\small 课程教材、可信检索与 AI 生成使用不同标识，来源不混淆。
\end{block}
\end{columns}
\vspace{4mm}
\begin{itemize}\setlength\itemsep{2.5mm}
  \item 危险请求：服务端返回 \texttt{CONTENT\_SAFETY\_BLOCKED}
  \item 用户 PDF：仅保存在本地私有目录，不进入 Git
  \item 重要事实：提示回到教材、教师或原始文献核验
\end{itemize}
