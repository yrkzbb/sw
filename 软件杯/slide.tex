\section{工程特色 / 创新}

\begin{frame}{可信不是附加项，而是生成链路的组成部分}
\begin{columns}[T,onlytextwidth]
\column{.32\textwidth}
\begin{block}{输入侧}
\small 浏览器与服务端双层拦截危险操作性请求，并防范提示注入。
\end{block}
\column{.32\textwidth}
\begin{block}{生成侧}
\small 资源进入结果池前审核；绝对化结论、论文与外链标记待复核。
\end{block}
\column{.32\textwidth}
\begin{block}{证据侧}
\small 课程教材、可信检索与 AI 生成使用不同标识，来源不混淆。
\end{block}
\end{columns}
\vspace{4mm}
\begin{itemize}\setlength\itemsep{2.5mm}
  \item 危险请求：服务端返回 \texttt{CONTENT\_SAFETY\_BLOCKED}
  \item 用户 PDF：仅保存在本地私有目录，不进入 Git
  \item 重要事实：提示回到教材、教师或原始文献核验
\end{itemize}
\end{frame}

\begin{frame}{创新价值来自四个“闭环连接”}
\begin{columns}[T,onlytextwidth]
\column{.48\textwidth}
\begin{block}{画像与行为连接}
\small 画像不是静态档案；学习行为持续改变内容难度、路径和推荐。
\end{block}
\begin{block}{生成与证据连接}
\small 每份资源都能区分课程依据、可信检索与 AI 扩展。
\end{block}
\column{.48\textwidth}
\begin{block}{多智能体与工程连接}
\small 独立调用、并行容错、轨迹保存，让协作可观察、可调试、可审核。
\end{block}
\begin{block}{资源与行动连接}
\small 资源进入阶段、Todo、掌握证据和复盘，不止停在“生成完成”。
\end{block}
\end{columns}
